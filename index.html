VOTING-SYSTEM.JS
// 线下投票系统
const voteOnTopic = (topicId, userId) => {
  const topic = getTopic(topicId);
  const user = getUser(userId);

  if (user.hasVoted(topicId)) {
    return { error: "已经投过票了" };
  }

  topic.votes += 1;
  user.addVote(topicId);

  return { success: true, newScore: topic.votes };
};
